3. Docker commands

3.1. run - start a container

docker run nginx

3.2. ps - list container

docker ps

#docker ps -a

3.3. stop - stop a container

docker stop 20871eb58c94
docker stop {{name}}

3.4. rm - remove a container

docker rm sleepy_carson

3.5. images - list images

docker images

3.6. rmi - remove images

docker rmi nginx

3.7. pull - download an image

docker pull nginx

3.8. exec - execute a command

docker run ubunto
#docker run ubunto sleep 20

docker exec hopeful_poincare cat /etc/hosts

4. Docker Run

4.1. Run - Tag

docker run redis

docker run redis:4.0

docker run kodekloud/simple-prompt-docker

4.2. Run - PORT mapping

docker run -p 80:5000 kodekloud/webapp

docker run -p 3306:3306 mysql

4.3. Run - Volume mapping

docker run mysql

docker stop mysql
docker rm mysql

docker run -v /opt/daradir:/var/lib/mysql mysql

4.4. Inspect container

docker inspect nostalgic_galois

4.5. Container logs

docker logs nostalgic_galois
